\begin{problem} 
Let $R$ be a ring, let $M$ be an $R$-module, and $N$ be a submodule of $M$.
\begin{enumerate}


\item Give an example of a free module $M$ and a submodule $N$ that is not free.

\item Give an example of a free module $M$ and a submodule $N$ that is free but $M/N$ is not free.

\item Suppose that $M$ is free with basis $B$. Suppose that $N$ is free with basis $C \subseteq B$. Show that $M/N$ is free with basis $\{ a + N \ | \ a \in B\smallsetminus C\}$.

\item Suppose that $N$ is free with basis $C$ and $M/N$ is free with basis $\{ a + N \ | \ a\in A\}$ for some subset $A\subset M$. Show that $M$ is free with basis $B= A \cup C$. 

\end{enumerate}

\end{problem}
