\begin{problem}
Let $R=\ZZ[X]$ and let $f:R^2 \to R$ be the linear transformation $f\left( \begin{bmatrix} p \\ q\end{bmatrix}\right) = 2p+Xq$. Show that there do not exist basis $B$ for $R^2$ and $C$ for $R$ and $r\in R$ for which $[f]_B^C = \begin{bmatrix} r & 0 \end{bmatrix}$.
\end{problem}


\begin{problem}
Let $K$ be a field, and $G$ be a finite subgroup of $K^\times$. Show that $G$ is cyclic.
\end{problem}
Note that $(\Z/p)^\times$ is cyclic, as a consequence of this.


\begin{problem} Prove\footnote{Hint: \begin{itemize}
\item Fix a basis $B$ for $F$. Let $\mathcal{P}= \{ A \subseteq B \ | \ RA \cap M \ \text{is free}\}$. Show that $\mathcal{P}$ has a maximal element.
\item Let $A\in \mathcal{P}$ be a maximal element, and suppose that there is some $b\in B\smallsetminus A$. Consider the map ${\pi\colon R(A\cup \{b\}) \cap M \to R}$ that maps an element to its $b$-coefficient.
\end{itemize}} that if $R$ is a PID, $F$ is a free module, and $M\subseteq F$ is a submodule (not necessarily finitely generated) then $M$ is free.
