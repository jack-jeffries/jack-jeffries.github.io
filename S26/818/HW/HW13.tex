\begin{problem}
Show that $\Q(\sqrt{2+\sqrt{2}})/\Q$ is a Galois extension of degree $4$ with Galois group that is a cyclic group of order $4$.	
\end{problem}

\


\begin{problem}
	Let $F \subseteq L$ be Galois field extension of degree 45. Prove there exists a unique intermediate field $E$ such that $[E : F] = 5$.
\end{problem}

\



\begin{problem} Let $f(x)=x^6-7\in \Q[x]$ and let $E$ be a splitting field of $f(x)$ over $\Q$.
Let $\omega=e^{2\pi i/6}$ be a primitive $6$th root of unity.
\begin{enumerate}[(a)]
\item Find, with justification, $[E:\Q]$.
\item Prove that $f(x)$ is irreducible over $\Q(\omega)$.
\item Prove that $\Aut(E/\Q)$ has a normal subgroup of order $3$.
\end{enumerate}
\end{problem}


\


\begin{problem}
Let $F$ be a field with $p^6$ elements for some prime $p$. Determine the number of subfields of $F$, and compute the number of elements in each such subfield.
\end{problem} 

\

\begin{problem} Let $F$ be a field of characteristic $p > 0$, $a \in F$, and consider the polynomial $f(x)=x^p -x-a \in F[x]$.

  \begin{enumerate}[(a)]

    \item Prove\footnote{Hint: If $\alpha$ is a root of
      $f(x)$,  consider $\alpha + j$ for $j \in \Z/p$.} that $f(x)$ is either irreducible over $F$ or it splits
      into distinct linear factors over $F$.

      \item Suppose $f(x)$ is irreducible over $F$ and let $L$ be a
        splitting field of $f$ over $F$.  Prove that the Galois group of $L$ over $F$ is cyclic.

\end{enumerate}
\end{problem}
