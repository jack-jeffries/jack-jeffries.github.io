	\section*{The Fundamental Theorem of Galois Theory}
	\begin{framed}
	\textsc{The Fundamental Theorem of Galois Theory:} 
	Let $L/F$ be a finite Galois extension. There is a bijection 
	\[ \begin{array}{ccccc} & \{ \text{fields} \ E \ | \ F\subseteq E \subseteq L \} &\longleftrightarrow & \{ \text{subgroups} \ H \leq \mathrm{Gal}(L/F) \} & \\
	& E & {\mapsto} & \mathrm{Gal}(L/E) & \\
	& L^H & {\mapsfrom} & H &
	\end{array}\]

Moreover, this bijective correspondence enjoys the following properties:
\begin{enumerate}
\item $\stackrel{\mathrm{Gal}(L/\star)}{\longrightarrow}$ and $\stackrel{L^{\star}}{\longleftarrow}$ each reverse the order of inclusion: 
\begin{itemize}
\item[] $E \subseteq E' \Longleftrightarrow \mathrm{Gal}(L/E') \leq \mathrm{Gal}(L/E)$; \ equivalently
\item[] $L^H \subseteq L^{H'} \Longleftrightarrow H' \leq H$.
\end{itemize}
\item $\stackrel{\mathrm{Gal}(L/\star)}{\longrightarrow}$ and $\stackrel{L^{\star}}{\longleftarrow}$ convert between degrees of extensions and indices of subgroups:  
\begin{itemize}
\item[] $[E:F] = [\mathrm{Gal}(L/F) : \mathrm{Gal}(L/E)]$; \ equivalently
\item[] $[\mathrm{Gal}(L/F) : H] = [L^H:F]$.
\end{itemize}
\item Normal subgroups correspond to intermediate fields that are Galois over $F$: 
\begin{itemize}
\item[] $E/F$ is Galois if and only if $\mathrm{Gal}(L/E) \trianglelefteq \mathrm{Gal}(L/F)$; \ equivalently
\item[] $N \trianglelefteq G$ if and only if $L^N/F$ is Galois.
\end{itemize}
In this case, $\mathrm{Gal}(E/F) = \mathrm{Gal}(L/F) / \mathrm{Gal}(L/E)$.
\end{enumerate}
		\end{framed}

			\itemA Let $L/F$ be a finite Galois extension.
	\begin{enumerate}
	\itema When is every intermediate field $E\subseteq L$ Galois over $F$?
	\itema If $[L:F]=n$ and $n=p^e m$ for some prime $p$ and positive integer $m$ that is not a multiple of~$p$, explain why there exists an intermediate field $E$ with $[E:F]=m$.
	\end{enumerate}
	
	
\solution{
	\begin{enumerate}
	\itema If and only if every subgroups is normal. This happens, for example, if $G$ is abelian.
	\itema Let $G= \mathrm{Gal}(L/F)$. Then by the Sylow Theorems, there is a subgroup $H\leq G$ of order $p^e$. Then $m = [G:H] = [L^H : F]$, so $E=L^H$ works.
	\end{enumerate}
	}
		
		\itemA Proof\footnote{Hint: Use Artin's Theorem and its Corollaries generously!} of Fundamental Theorem: 
		\begin{enumerate}
		\itema Show that the maps $\stackrel{\mathrm{Gal}(L/\star)}{\longrightarrow}$ and $\stackrel{L^{\star}}{\longleftarrow}$ give the bijection as stated in the Theorem.
		\itema Prove part (1) of The Fundamental Theorem of Galois Theory. 
		\itema Prove part (2) of The Fundamental Theorem of Galois Theory.
		\end{enumerate}
		
		\solution{
		\begin{enumerate}
		\itema First, we note that these are well-defined: for $E\subseteq L$, $L/E$ is indeed Galois by a Corollary of Artin's Theorem, and $\mathrm{Gal}(L/E)$ is a subgroup of $\mathrm{Gal}(L/F)$. Also, for $H\subseteq G$, $L^H$ is a subfield of $L$ by an exercise, and $L^H$ contains $F$ since every element of $H$ fixes $F$.
		
		
		Let $F\subseteq E \subseteq L$. Then $E$ maps to $\mathrm{Gal}(L/E)\leq \mathrm{Gal}(L/F)$. By a Corollary of Artin's Theorem, we have $L^{\mathrm{Gal}(L/E)} = E$. This shows that one composition is the identity.
		
		Let $H\leq \mathrm{Gal}(L/F)$. Then $H$ maps to $L^H$. This maps to $\mathrm{Aut}(L/L^H)$ which is $H$ by Artin's Theorem. This shows that the other composition is the identity.
		
		\itema This follows from the definitions: if $E\subseteq E'$, then any automorphism that fixes $E'$ definitely fixes $E$, so $\mathrm{Gal}(L/E') \leq \mathrm{Gal}(L/E)$. For the converse, we write $E=L^H$ and $E'=L^{H'}$; if $H' \leq H$, then anything fixed by all elements of $H$ is fixed by all elements of $H'$, so $E=L^H \subseteq L^{H'}=E'$.
		
		\itema By definition of Galois, we have $[L:F] = |\mathrm{Gal}(L/F)|$. Since $L/E$ is also Galois, we have $[L:E] = |\mathrm{Gal}(L/E)|$. Then
		\[ [E:F] = \frac{[L:F]}{[L:E]} = \frac{|\mathrm{Gal}(L/F)|}{ |\mathrm{Gal}(L/E)|} = [ \mathrm{Gal}(L/F) : \mathrm{Gal}(L/E)].\]
		\end{enumerate}
		}
		
		
	

	
		
	\end{enumerate}
