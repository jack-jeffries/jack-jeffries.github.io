Here are some old qualifying exam problems you are already ready to solve.


\vspace{2em}




\begin{problem} Let $R$ be a commutative integral domain and $M$ an $R$-module. A subset $S$ of $M$ is called a {\em maximal linearly independent} subset of $M$ if $S$ is 
linearly independent and any subset of $M$ properly containing $S$ is linearly dependent.

\begin{enumerate}[(a)]

\item Let $T$ be a linearly independent subset of $M$. Prove that $T$ is contained in some maximal linearly independent subset of $M$.

\item Let $T$ be a linearly independent subset of $M$ and let $N$ be the $R$-submodule of $M$ generated by $T$. Prove that $T$ is a maximal linearly independent subset if and only
if $M/N$ is torsion. (Recall that an $R$-module $P$ is called ``torsion'' if for each $p \in P$, there is a $r \in R$ such that $r \ne 0$ and $rp = 0$.)

\end{enumerate}
\end{problem}



\begin{problem} Let $R$ be a nonzero commutative ring and suppose $p: R^m \to R^n$ is a surjective homomorphism of $R$-modules for some nonnegative integers $m$ and $n$. 

\begin{enumerate}[(a)]

\item Prove that if $R$ is a field, then $m \geq n$.

\item Prove that if $R$ is any commutative ring, then $m \geq n$. 

\end{enumerate}

\end{problem}



\begin{problem}
Let $A = \begin{bmatrix} 6 & 4 & 2 \\ 2 & 7 & 4 \end{bmatrix} \in \mathrm{Mat}_{2 \times 3}(\Z)$. 

\begin{enumerate}[(a)]

\item Find invertible matrices $P$ and $Q$ such that $B := PAQ$ is diagonal (i.e., $b_{i,j} = 0$ for 
all $i \ne j$). 

\item Find all integer solutions of $A \begin{bmatrix} x \\ y \\ z \end{bmatrix} = \begin{bmatrix} 
0 \\ 0 \end{bmatrix}$.


\end{enumerate}
\end{problem}


\begin{problem}
Let $F$ be a field, $V$ an $F$-vector space, and $W$ a subspace of $V$. A subspace $U$ of $V$ is called a {\em complement} of
  $W$  in $V$ if $U \cap W = \{0\}$ and $U + W = V$. 

  \begin{enumerate}[(a)]
     \item Prove that for every $V$ and $W$ as above, $W$ has at least one complement in $V$.
     \item Prove that if $U$ is a complement of $W$ in $V$ and $V$ is finite dimensional, then $\dim_F(V)=\dim_F(W)+\dim_F(U)$.
     \end{enumerate}\end{problem}

\begin{problem} Let $R$ be a Euclidean domain, $A$ an $m \times n$ matrix with elements from $R$, and $A^T$ the transpose matrix of $A$.
         Let $\coker(A)$ denote the quotient of $R^m$ by the submodule generated by the columns of $A$. The {\em torsion} submodule of an $R$-module
         $M$ is the submodule \[\{m \in M \mid rm = 0 \text{ for some $r \ne 0$}\}.\] (It is indeed a submodule and you need not prove this.) 

         \begin{enumerate}[(a)]
       \item  Prove that the torsion submodules of $\coker(A)$ and $\coker(A^T)$ are isomorphic.

         \item Prove that the modules $\mathrm{coker}(A)$ and $\mathrm{coker}(A^T)$ are isomorphic if and only if $m = n$.

        \end{enumerate}
\end{problem}


\begin{problem} Let $I$ be an ideal in a commutative ring $R$, let $M$ and $N$
  be $R$-modules and let $f : M \to N$ be an $R$-module homomorphism.
  \begin{enumerate}[(a)]
    \item Prove there is a unique $R$-module homomorphism $f : M/IM
      \to N/IN$ such that $f \circ p = q \circ f$, where $p : M \to
      M/IM$ and $q : N \to N/IN$ are the canonical quotient maps.

      \item Prove that if $I^2 = 0$ and $f$ is surjective, then so is $f$. (Recall that $I^2$ is the ideal generated by all elements of the form $ab$, where $a, b\in I$ .)
    \end{enumerate}
    \end{problem}
