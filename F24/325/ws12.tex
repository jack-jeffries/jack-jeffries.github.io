	\section*{Properties of sequences and convergence \S2.1}
	
	

\begin{framed}
\noindent \textsc{Definition 12.1:} Let $\{a_n\}_{n=1}^{\infty}$ be a sequence.
\begin{enumerate}
\item We say $\{a_n\}_{n=1}^\infty$ is  \textbf{increasing}\index{increasing} if for all $n \in \N$ we have $a_n \leq a_{n+1}$.
\item We say $\{a_n\}_{n=1}^\infty$ is  \textbf{decreasing}\index{decreasing} if for all $n \in \N$, we have $a_n \geq a_{n+1}$.\item We say $\{a_n\}_{n=1}^\infty$ is  \textbf{monotone}\index{monotone} if it is either decreasing or increasing.
\item We say $\{a_n\}_{n=1}^\infty$ is  \textbf{strictly increasing}\index{strictly increasing} if for all $n \in \N$, $a_n < a_{n+1}$.
\end{enumerate}
 I leave the definition of \textbf{strictly decreasing}\index{strictly decreasing} and \textbf{strictly monotone}\index{strictly monotone} to your imaginations.



\end{framed}

\begin{enumerate}
\item For each of the following sequences which of the following adjectives apply: bounded above, bounded below, bounded, (strictly) increasing, (strictly) decreasing, (strictly) monotone?
\begin{enumerate}
\item $\{ \frac{1}{n} \}_{n=1}^\infty$
\item The Fibonacci sequence $\{ f_n \}_{n=1}^\infty$ where $f_1=f_2=1$ and $f_n= f_{n-1} + f_{n-2}$ for $n\geq 3$.
\item $\{(-1)^n \}_{n=1}^\infty$
\item $\{ 5 + (-1)^n \frac{1}{n}  \}_{n=1}^\infty$.
\end{enumerate}

\

\item Prove or disprove: Every increasing sequence is bounded above.

\

\item Prove or disprove: Every increasing sequence is bounded below.

\

\item Prove or disprove: Every bounded sequence is convergent.

\

\item Suppose that $\{a_n\}_{n=1}^\infty$ is a sequence that is bounded above by $1000$ and below by $-1000$. Show that the sequence $\displaystyle \left\{\frac{a_n}{n}\right\}_{n=1}^\infty$ converges to $0$.

Suggestion: First start the way we always do when showing a sequence converges. Then see if you can use the hypothesis that $\{a_n\}_{n=1}^\infty$ is bounded in a useful way.


\end{enumerate}
