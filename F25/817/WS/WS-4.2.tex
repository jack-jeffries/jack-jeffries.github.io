	\section*{Normal subgroups}
	
	

\begin{framed}
\textsc{Definition:} A subgroup $N$ of a group $G$ is \textbf{normal} if $gNg^{-1} = N$ for all $g\in G$. We write $N\trianglelefteq G$ to indicate that $N$ is a normal subgroup of $G$.


\textsc{Lemma:} Let $N$ be a subgroup of a group $G$. The following are equivalent:
\begin{enumerate}[(a)]
\item $N$ is a normal subgroup of $G$.
\item For all $g\in G$, $gNg^{-1} \subseteq N$.
\item 
\end{enumerate}
\end{framed}
