Here are some old qualifying exam problems you are already ready to solve.


\vspace{2em}

\begin{problem}
\begin{itemize}
\item[(a)] Let $G$ be a simple group of order 60.
Determine the number of elements of $G$ of order 5.
\item[(b)] Show that there is no simple group of order 30.
\end{itemize}
\end{problem}

\medskip

\begin{problem}

\item A non-abelian group of order $27$.

\begin{enumerate}[(a)]

\item Prove there exists a non-abelian group of $27$.   ({\em Hint:} Use a semi-direct product.)

\item Find, with justification, a presentation of the group you found in part (a).

\end{enumerate}
\end{problem}

\medskip

\begin{problem}
Determine, with justification, all the isomorphism classes of groups of order $45$.
\end{problem}

\medskip

\begin{problem}
 Let $G = A_7$ and $S$ be the set of all elements of $G$ of order $7$. Prove that $S$ is not a conjugacy class of $G$.
 \end{problem}

\medskip

\begin{problem} Suppose $H$ is a subgroup of a group $G$ and $[G : H] = 7$. 


\begin{enumerate}[(a)]

\item Prove $G$ contains a normal subgroup $N$ such that $N \subseteq H$ and $[G : N] \leq 7$!.

\item 
Prove $7!$ is the best possible bound for the previous part â€” i.e., prove there is a group $G$ and a subgroup $H$ with $[G : H] = 7$ such that for every normal subgroup
$N$ of $G$ with $N \subseteq H$, we have $[G : N] \geq 7!$.
\end{enumerate}
 \end{problem}
 
 \medskip

\begin{problem} Consider the following statement, which can be viewed as a converse to Lagrange's theorem: 
  \begin{quote}
    Let $G$ be a finite group of order $n$ and $d$ a positive divisor of $n$. Then $G$ has a subgroup of order $d$.
  \end{quote}
  
\begin{enumerate}[(a)]
\item Give a counterexample (with complete justification) to the above statement.
\item Prove that if one adds the hypothesis that $G$ is abelian, then the above statement is true.
\end{enumerate}
 \end{problem}
 
 \medskip

\begin{problem} Let $G$ be a group and $K$ a finite cyclic normal subgroup of $G$. Prove that $G' \subseteq C_G(K)$, where $G'$ is the commutator subgroup of $G$ and
$C_G(K) = \{g \in G \mid gk = kg \text{ for all $k \in K$ } \}$. (Hint: Consider an appropriate action of $G$ on $K$.)
 \end{problem}
 
 \medskip

\begin{problem} Prove that no group of order $150$ is simple. 
 \end{problem}
