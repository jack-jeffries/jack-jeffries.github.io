	{\LARGE  Some old qualifying exam questions on rings}
\end{center}


Here are some old qualifying exam problems you are already ready to solve.



  \begin{problem}
  Let $R$ be a commutative ring and $x$ an indeterminant. Prove that $R[x]$ is a principal ideal domain (PID) if and only if $R$ is a field.
\end{problem}
  

  \begin{problem} Let $d$ be a square free integer and $\Q(\sqrt{d})$ the subring of $\C$ defined by $\Q(\sqrt{d})=\{a+b \sqrt{d} \mid a,b \in \Q\}$.
Show that there is a ring isomorphism $\Q[x]/(x^2 - d) \cong \Q(\sqrt{d})$.
\end{problem}



  \begin{problem}
    \begin{enumerate}[(a)]
    \item Prove that a finite integral domain must be a field.
      \item Prove that if $R$ is a commutative ring and $P \subseteq R$ is a prime ideal such that $P$ has finite
index as a subgroup of $(R, +)$, then $P$ is a maximal ideal. Give an example to show that this implication may fail if the finite index assumption is dropped.
\end{enumerate}
\end{problem}
  

  \begin{problem}
  Let $R$ be a commutative ring, and set $I = \{r \in R \mid r^n =0 \text{ for some integer $n \geq 1$}\}$. Prove that following assertions.
  \begin{enumerate}[(a)]
    \item $I$ is an ideal in $R$.
\item If $R/I$ is a field, then each element of $R$ is either a unit or
  in $I$.
 \end{enumerate}
\end{problem}


  \begin{problem} Let $I$ be a nonzero ideal of the ring of Gaussian integers $\Z[i]$. Prove that the quotient ring $\Z[i]/I$ is finite.
  \end{problem}

  \begin{problem} On UFDs.
  \begin{enumerate}[(a)]
    \item Prove that in a UFD a nonzero element $p$ is irreducible if and only if the ideal it generates $(p)$  is a prime ideal.
\item Prove that $\Z[ \sqrt{-5}]$ is not a UFD.
\end{enumerate}
\end{problem}
