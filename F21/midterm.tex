\maketitle

For this exam, you may use the lecture notes, your own notes, and the homework assignments, but you should not use other sources, including your classmates. However, if you plan to take a written algebra comp, I recommend that you first attempt the problems without notes, and then complete them with notes later. Turn in solutions to \emph{five} of the problems below.

\


\begin{enumerate}

\item \begin{enumerate}
\item Let $P$ be a poset. Show that every morphism in $\PO(P)$ is both monic and epic.
\item Give an example of a poset $P$, a morphism $\alpha\in \PO(P)$, and a functor $F:\PO(P) \to \mathscr{C}$ to some category $\mathscr{C}$ such that $F(\alpha)$ is neither monic nor epic.
\end{enumerate}

\

\item Let $K$ be a field, and $R=M_n(K)$ be the ring of $n\times n$ matrices over $K$. Let $M$ be the right $R$-module of $1\times n$ row vectors and $N$ be the left $R$-module of $n\times 1$ column vectors. Observe that $M$ and $N$ are also $K$-vectorspaces.
\begin{enumerate}
\item Show that $M\otimes_K N \cong R$ (as abelian groups).
\item Show that $M\otimes_R N \cong K$ (as abelian groups).

\end{enumerate}

\

\item Let $K$ be a field. Show that any additive functor $F:\Vs{K}\to\Vs{K}$ is exact.

\

\item Let $\Set^\infty$ be the full subcategory of $\Set$ consisting of all infinite sets. Let $F:\Set^\infty \to \Set^\infty$ be the functor that on objects is given by the rule $F(S)=S\times S$, and on morphisms is given by $F(f)=(f,f)$. Show that there is no natural isomorphism $\eta: F\Rightarrow 1_{\Set^{\infty}}$.

\

\item Give examples of $\Z$-modules with the following sets of properties, and justify.
\begin{enumerate}
\item Injective, but not flat.
%\item Flat, but neither injective nor projective.
\item Flat and injective, but not projective.
\end{enumerate}

\

\item Let $\phi:R\to S$ be a ring homomorphism. Let $N$ be an $R$-module, and $M$ be an $S$-module. Let $F:\Mod{S}\to \Mod{R}$ be restriction of scalars along $\phi$, and $G:\Mod{R} \to \Mod{S}$ be the functor $\Hom_R(S,-)$. Show that $\Hom_R(F(M),N) \cong \Hom_S( M, G(N))$.

\

\item Let $R$ be a commutative ring, and $M,N$ be $R$-modules.
\begin{enumerate}
\item Show that if $M$ and $N$ are projective, then $M\otimes_R N$ is projective.
\item Show that if $M$ and $N$ are flat, then $M\otimes_R N$ is flat.
\end{enumerate}

\end{enumerate}
