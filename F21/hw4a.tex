\begin{enumerate}


\item Find two different composition series for the ring $R=\Z[x] / (6,x^2)$ as a module over itself.

\

\item The purpose of this problem is to show that the ring
\[ R = \begin{bmatrix} \Z & 0 \\ \Q & \Q \end{bmatrix} = \left\{ \begin{bmatrix} a &  0 \\ b & c \end{bmatrix} \ \Bigg| \ a\in \Z, \ b,c\in \Q\right\}\]
is left Noetherian but not right Noetherian.
\begin{enumerate}
\item Let $I$ be a left ideal. Show that if there is some $\begin{bmatrix} a & 0 \\ b & c\end{bmatrix}\in I$ with $a\neq 0$, then for some ideal $J\subseteq \Z$,
 \[ I=\begin{bmatrix} J & 0 \\ \Q & 0\end{bmatrix} \quad\textrm{or } \quad I=\begin{bmatrix} J & 0 \\ \Q & \Q\end{bmatrix}.\]

\item Show that the left ideals contained in $\begin{bmatrix} 0 & 0 \\ \Q & \Q \end{bmatrix}$ are exactly the $\Q$-vector subspaces of $\begin{bmatrix} 0 & 0 \\ \Q & \Q \end{bmatrix}$. Conclude that $R$ is left Noetherian.
\item Show that for every $n\in \N$, the subset of $R$ given by $\begin{bmatrix} 0 & 0 \\ \frac{1}{n}\Z & 0 \end{bmatrix}$ is a right ideal. Conclude that $R$ is not right Noetherian.
\end{enumerate}


\



\item\label{prob} Computations with Schur's Lemma.
\begin{enumerate}
\item Let $D$ be a division ring and $n\geq 1$ be an integer. Show\footnote{Hint: let $x\in D^{\op}$ correspond to the map given by multiplication on the right by $x$.} that $\End_{\Mat_n(D)} (D^n) \cong D^{\op}$.
\item Let $n\geq 3$, and let $D_{2 n}$ be the dihedral group of order $2n$. Show that the standard representation of $D_{2n}$ on $V=\R^2$ acting by rotations and reflections is simple.
\item With $V$ as in the previous part, show that $\End_{\R[D_{2n}]}(V) \cong \R$ as rings.
\end{enumerate}

\


\item The goal of this problem is to find every irreducible
  $\R$-linear representation (up to isomorphism) of the dihedral group $D_{8}$ of order $8$. 

\begin{enumerate}

\item Prove\footnote{Recall that a $1$-dimensional representation of $G$ is a homomorphism from $G$ to $\R^\times$, and that any such map factors through the abelianization of $G$. It might be helpful to note that the derived subgroup of $D_8$ is $\langle r^2 \rangle$ where $r$ is the ``rotation'' and the abelianization of $D_8$ is isomorphic to $C_2 \times C_2$.}
 there are $4$ irreducible representations $M_1, M_2, M_3, M_4$ whose underlying $\R$-vector space is just $\R$ and that are pairwise nonisomorphic.
\item Let $M_5$ be $\R^2$ with the standard action of $D_8$ (as in problem~\ref{prob}). Prove that $M_1, \dots, M_5$ are the only irreducible $\R$-linear representations of $D_8$ up to isomorphism.
\end{enumerate}

\


\item Let $Q=\{\pm 1,\pm i,\pm j,\pm k\}$ denote the group of quaternions.


\begin{enumerate} 

\item Find the Artin-Wedderburn decomposition of each of $\C[Q]$ and $\C[D_8]$ and use these to prove there
is a ring isomorphism $\C[Q] \cong \C[D_8]$. 

\def\bH{{\mathbb H}}

\item Find\footnote{Hint: Use the previous problem to find the Artin-Wedderburn decomposition of $\R[D_8]$. To find the decomposition of $\R[Q]$, note that the group $Q$ acts $\R$-linearly on the left on the division ring of real quaternions $\bH$, via the identification of $Q$ as the subgroup $\{ \pm 1, \pm i, \pm j, \pm k\}$ of $\bH^\times$,  and thus $\bH$ is a left $\R[Q]$-module. Thus, there is a surjective ring map $\R[Q]\to \bH$. } the Artin-Wedderburn decomposition of each of $\R[Q]$ and $\R[D_8]$  and use these to prove these rings are not isomorphic.
