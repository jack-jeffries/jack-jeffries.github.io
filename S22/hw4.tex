\begin{enumerate}

\item Let $K$ be a field, and $R$ be a positively graded $K$-algebra. Let $M$ be\footnote{Note that we are not assuming that $M$ is finitely generated.} an $\mathbb{N}$-graded $R$-module.
\begin{enumerate}
\item Show that $(R_+)M = M$ implies $M=0$.
\item Show that for a subset $S\subset M$, $M$ is generated by $S$ as an $R$-module if and only if $M/(R_+)M$ is generated by the images of the elements of $S$ as a $K$-vector space.
\end{enumerate}

\ 


\item Compute the dimension of each of the following rings $R$, where $K$ is a field and $x,y,z,u,v$ are indeterminates:
\smallskip
\smallskip
\begin{enumerate}
\item $\displaystyle R=\frac{K[x,y,z]}{(x^3,x^2y,xyz)}$.
\smallskip
\item $\displaystyle R=K[x^2 u, xyu, y^2 u, x^2 v, xyv, y^2v] \subseteq K[x,y,u,v]$.
\smallskip
\item $\displaystyle R=\frac{K[x,y,z,u,v]}{(x^3 u^2 + y^3 uv + z^3 v^2)}$.
\smallskip
\item $\displaystyle R=\frac{K[x,y,u,v]}{(u^3 - xy, v^5 - x^2 u - y^3)}$.
\end{enumerate}

\medskip
\

\item Let $K$ be a field, and $R\subseteq S$ be a module-finite inclusion of finitely generated $K$-algebras that are both domains\footnote{Note that we are not assuming that $R$ is normal.}. Show that for any $\mathfrak{q}\in \mathrm{Spec}(S)$, $\mathrm{height}(\mathfrak{q}) = \mathrm{height}(\mathfrak{q} \cap R)$.

\


\item Let $\psi: R \hookrightarrow S$ be an algebra-finite inclusion of rings. 
\begin{enumerate}
\item Show that if $R$ is a domain, then $\mathrm{im}(\psi^*)$ contains a nonempty open subset of $\mathrm{Spec}(R)$.
\item Show that\footnote{First show that each minimal prime $\mathfrak{p}$ is in the image of $\mathrm{im}(\psi^*)$, so $\mathfrak{p} S \cap R= \mathfrak{p}$. To see this, you may want to consider the localization of the map $\psi$ at $(R\smallsetminus \mathfrak{p})$.} for every minimal prime $\mathfrak{p}$ of $R$, $\mathrm{im}(\psi^*)$ contains a nonempty open subset of $V(\mathfrak{p})$.
\end{enumerate}



\


\item Let $X= \begin{bmatrix} x_{11} & x_{12} & x_{13} \\ x_{21} & x_{22} & x_{23} \end{bmatrix}$ be a $2\times 3$ matrix of indeterminates over $\mathbb{C}$ and  $R=\mathbb{C}[X]$. Let $I$ be the ideal of $2\times 2$ minors of $X$. Show\footnote{Hint: You may want to use the problem (1) to show that that the map is module-finite.} that $\mathbb{C}[x_{11}, x_{12}-x_{21}, x_{13}-x_{22}, x_{23}]$ is a Noether normalization for $R/I$, and conclude that the height of $I$ is two.

\


\end{enumerate}
